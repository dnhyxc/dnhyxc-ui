# Button

自定义 Button

```tsx
import { Button } from '@dnhyxc-ui/base'

export default function App() {
  return (
    <div style={{ textAlign: 'center' }}>
      <Button />
    </div>
  )
}
```

带欢迎语

```tsx
import { Button } from '@dnhyxc-ui/base'

export default function App() {
  return (
    <div style={{ textAlign: 'center' }}>
      <Button size={65} welcome="下午好，有什么可以帮你吗？" />
    </div>
  )
}
```
